\item $\vdash 0 \to A$ % (a)
\mdseries
\begin{center}
	\R{$0 \vdash 0$}
	\botE{$ 0 \vdash A$}
	\toI{$\vdash 0 \to A$}
	\DisplayProof
\end{center}

\bfseries
\item $\vdash A \lor ( A \land  B) \iff A$ % (b)
\mdseries
\begin{center}
	\R{$A \lor ( A \land  B) \vdash A \lor ( A \land  B)$}
	\R{$A \lor ( A \land  B), A \vdash A$}
	\R{$A \lor ( A \land  B), A \land B \vdash A \land B$}
	\andEl{$A \lor ( A \land  B), A \land B \vdash A$}
	
	\orE{$A \lor ( A \land  B) \vdash A$}
	\toI{$\vdash A \lor ( A \land  B) \to A$}
	
	\R{$A \vdash A$}
	\orIl{$A \vdash A \lor (A \land B)$}
	\toI{$\vdash A \to A \lor (A \land B)$}
	
	\andI{$\vdash A \lor ( A \land  B) \iff A$}
	
	\DisplayProof
\end{center}

\bfseries   % (c)
\item $A \land (A \lor B) \iff A$
\mdseries
\begin{center}
	\R{$A \land (A \lor B) \vdash A \land (A \lor B)$}
	\andEl{$A \land (A \lor B) \vdash A$}
	\toI{$ \vdash A \land (A \lor B) \to A$}
	
	\R{$ A \vdash A$}
	\R{$ A \vdash A$}
	\orIl{$A \vdash A \lor B$}
	\andI{$A \vdash A \land (A \lor B)$}
	\toI{$ \vdash A \to (A \land (A \lor B))$}
	
	\andI{$A \land (A \lor B) \iff A$}
	\DisplayProof
\end{center}

	\toI{$A \to B \land C \vdash A \to B$}

	\toI{$A \to B \land C \vdash A \to C$}
	
	\andI{$A \to B \land C \vdash (A \to B) \land (A \to C)$}

	\toI{$(A \to B) \land (A \to C) \vdash A \to B \land C$}

\bfseries
\item $\lnot (A \lor B) \iff \lnot A \land \lnot B$
\mdseries
\begin{center}
	\R{$(A \lor B) \to 0, A \vdash (A \lor B) \to 0$}
	\R{$(A \lor B) \to 0, A \vdash A$}
	\orIl{$(A \lor B) \to 0, A \vdash A \lor B$}
	\toE{$(A \lor B) \to 0, A \vdash 0$}
	\toI{$(A \lor B) \to 0 \vdash A \to 0$}
	
	\R{$(A \lor B) \to 0, B \vdash (A \lor B) \to 0$}
	\R{$(A \lor B) \to 0, B \vdash B$}
	\orIr{$(A \lor B) \to 0, B \vdash A \lor B$}
	\toE{$(A \lor B) \to 0, B \vdash 0$}
	\toI{$(A \lor B) \to 0 \vdash B \to 0$}
	
	\andI{$(A \lor B) \to 0 \vdash (A \to 0) \land (B \to 0)$}
	\toI{$\vdash ((A \lor B) \to 0) \to (A \to 0) \land (B \to 0)$}
	\DisplayProof
	
	\R{$(A \to 0) \land (B \to 0), A \lor B \vdash A \lor B$}
	
	\R{$(A \to 0) \land (B \to 0), A \lor B, A \vdash (A \to 0) \land (B \to 0)$}
	\andEl{$(A \to 0) \land (B \to 0), A \lor B, A \vdash A \to 0$}
	\R{$(A \to 0) \land (B \to 0), A \lor B, A, \vdash A$}
	\toE{$(A \to 0) \land (B \to 0), A \lor B, A \vdash 0$}
	
	\R{$(A \to 0) \land (B \to 0), A \lor B, B \vdash (A \to 0) \land (B \to 0)$}
	\andEr{$(A \to 0) \land (B \to 0), A \lor B, B \vdash B \to 0$}
	\R{$(A \to 0) \land (B \to 0), A \lor B, B, \vdash B$}
	\toE{$(A \to 0) \land (B \to 0), A \lor B, B \vdash 0$}
	
	\orE{$(A \to 0) \land (B \to 0), A \lor B \vdash 0$}
	\toI{$(A \to 0) \land (B \to 0) \vdash (A \lor B) \to 0$}
	\toI{$\vdash (A \to 0) \land (B \to 0) \to ((A \lor B) \to 0)$}
	\DisplayProof
\end{center}

\bfseries
\item $A \iff (A \to 0) \to 0$ % (g)
\mdseries
\begin{center}
	\R{$A, A \to 0 \vdash A$}
	\R{$A, A \to 0 \vdash A \to 0$}
	\toE{$A, A \to 0 \vdash 0$}
	\toI{$A \vdash (A \to 0) \to 0$}
	\toI{$\vdash A \to ((A \to 0) \to 0)$}
	
	\R{$(A \to 0) \to 0, A \vdash A$}
	
	\R{$(A \to 0) \to 0, A \to 0 \vdash (A \to 0) \to 0$}
	\R{$(A \to 0) \to 0, A \to 0 \vdash A \to 0$}
	\toE{$(A \to 0) \to 0, A \to 0 \vdash 0$}
	\botE{$(A \to 0) \to 0, A \to 0 \vdash A$}
	
	\case{$(A \to 0) \to 0 \vdash A$}
	\toI{$\vdash ((A \to 0) \to 0) \to A$}
	
	\andI{$A \iff (A \to 0) \to 0$}
	\DisplayProof
	
\end{center}

\bfseries
\item $\vdash ((A \lor B) \to 0) \iff (A \to 0) \land (B \to 0)$ % (h)
\mdseries
\begin{center}
	\R{$(A \lor B) \to 0, A \vdash A$}
	\orIl{$(A \lor B) \to 0, A \vdash A \lor B$}
	\R{$(A \lor B) \to 0, A \vdash (A \lor B) \to 0$}
	\toE{$(A \lor B) \to 0, A \vdash 0$}
	\toI{$(A \lor B) \to 0 \vdash A \to 0$}
	
	\R{$(A \lor B) \to 0, B \vdash B$}
	\orIr{$(A \lor B) \to 0, B \vdash A \lor B$}
	\R{$(A \lor B) \to 0, B \vdash (A \lor B) \to 0$}
	\toE{$(A \lor B) \to 0, B \vdash 0$}
	\toI{$(A \lor B) \to 0 \vdash B \to 0$}
	
	\andI{$(A \lor B) \to 0 \vdash (A \to 0) \land (B \to 0)$}
	\toI{$\vdash ((A \lor B) \to 0) \to (A \to 0) \land (B \to 0)$}
	\DisplayProof
	
	\R{$(A \to 0) \land (B \to 0), A \lor B \vdash A \lor B$}
	
	\R{$(A \to 0) \land (B \to 0), A \lor B, A \vdash (A \to 0) \land (B \to 0)$}
	\andEl{$(A \to 0) \land (B \to 0), A \lor B, A \vdash A \to 0$}
	\R{$(A \to 0) \land (B \to 0), A \lor B, A \vdash A$}
	\toE{$(A \to 0) \land (B \to 0), A \lor B, A \vdash 0$}
	
	\R{$(A \to 0) \land (B \to 0), A \lor B, B \vdash (A \to 0) \land (B \to 0)$}
	\andEr{$(A \to 0) \land (B \to 0), A \lor B, B \vdash B \to 0$}
	\R{$(A \to 0) \land (B \to 0), A \lor B, B \vdash B$}
	\toE{$(A \to 0) \land (B \to 0), A \lor B, B \vdash 0$}
	
	\orE{$(A \to 0) \land (B \to 0), A \lor B \vdash 0$}
	\toI{$(A \to 0) \land (B \to 0) \vdash (A \lor B) \to 0$}
	\toI{$\vdash (A \to 0) \land (B \to 0) \to (A \lor B) \to 0$}
	\DisplayProof
	
	\iffalse
	\predet{$\vdash ((A \lor B) \to 0) \to (A \to 0) \land (B \to 0)$}
	\predet{$\vdash (A \to 0) \land (B \to 0) \to (A \lor B) \to 0$}

	\fi
\end{center}

\bfseries % (i)
\item $(A \to B) \iff (\lnot B \to \lnot A)$
\mdseries
\begin{center}
	\R{$A \to B, B \to 0, A \vdash B \to 0$}
	\R{$A \to B, B \to 0, A \vdash A \to B$}
	\R{$A \to B, B \to 0, A \vdash A$}
	\toE{$A \to B, B \to 0, A \vdash B$}
	\toE{$A \to B, B \to 0, A \vdash 0$}
	\toI{$A \to B, B \to 0 \vdash A \to 0$}
	\toI{$A \to B \vdash (B \to 0) \to (A \to 0)$}
	\toI{$\vdash (A \to B) \to ((B \to 0) \to (A \to 0))$}
	\DisplayProof
	
	\R{$(B \to 0) \to (A \to 0), A, B \vdash B$}
	\R{$(B \to 0) \to (A \to 0), A, B \to 0 \vdash (B \to 0) \to (A \to 0)$}
	\R{$(B \to 0) \to (A \to 0), A, B \to 0 \vdash B \to 0$}
	\toE{$(B \to 0) \to (A \to 0), A, B \to 0 \vdash A \to 0$}
	\R{$(B \to 0) \to (A \to 0), A, B \to 0 \vdash A$}
	\toE{$(B \to 0) \to (A \to 0), A, B \to 0 \vdash 0$}
	\botE{$(B \to 0) \to (A \to 0), A, B \to 0 \vdash B$}
	\case{$(B \to 0) \to (A \to 0), A \vdash B$}
	\toI{$(B \to 0) \to (A \to 0) \vdash A \to B$}
	\toI{$\vdash ((B \to 0) \to (A \to 0)) \to (A \to B)$}
	\DisplayProof
\end{center}
